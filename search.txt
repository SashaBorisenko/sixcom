#include <iostream>
#include <fstream>
#include <string>
using namespace std;

void search(string search)
{
	ifstream searchfile;
	searchfile.open("search.txt");
	if (searchfile.is_open())
	{
		while (!searchfile.eof())
		{
			string line;
			while (getline(searchfile, line)) {
				if (line.find(search, 0) != string::npos)
				{
					cout << line << endl;
				}
			}
			searchfile.close();
		}
	}
	else
		cout << "not found" << endl;
}


int main()
{
	search("0546578357");
	system ("pause");
	return 0;
}
